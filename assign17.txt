CW	[ 

			Code
			
		]
	=> 
	
		CFN [ Code ][ 0 ][ 1 ]
		CWN [ Code ][ 0 ][ 1 ]
		
	| always



CFN [

			while (Cond)
			{
				Body
			}
			Rest
			
		][ X ][ Y ]
	=>
	
			bool While_X_Y()
			{
				CFN[ Body ][ X_Y ][ 1 ]
				CWN[ Body ][ X_Y ][ 1 ]
				return (Cond && While_X_Y());
			}
			CFN[ Rest ][ X ][ Y+1 ]
			
	| always



CWN [

			while (Cond)
			{
				Body
			}
			Rest
			
		][ X ][ Y ]
	=>
	
			if (Cond)
			{
			While_X_Y();
			}
			CWN[ Rest ][ X ][ Y+1 ]
			
	| always
